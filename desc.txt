Сервис будет иметь два эндпойнта:
POST /api/auth
GET /api/servers

POST /api/auth
Request:
- Content-Type: application/json
- {
    "login": "<your_login>",
    "password": "<your_pass>"
}

Responses:
 - 200 OK
 Content-Type: application/json
 {
    "jwt": "JWT_token"
 }

 - 400 Bad Request
    {
        "error": "invalid password"
    },
    {
        "error": "invalid json"
    }
 - 500 Internal Error

GET /api/servers
Request:
- Authorization: Bearer <jwt_token>

Responses:
- 200 OK
    [
        list of servers name
    ]

- 401:
- {
    "error": "Omit Authorization header"
},
{
    "error": "Invalid Authorization scheme"
},
{
    "error": "invalid jwt"
}
- 500 Internal Error
